\title{Find an entry in a matrix}
\author{Crichton Ogle}

\begin{document}

\begin{question} Suppose
\[
A = \begin{bmatrix}
1 & 2 & -1 & 4\\
0 & 0 & 1 & 3\\
0 & 1 & 0 & 2
\end{bmatrix}.\]
Then $A(1,1) = \answer{1}$ and $A(1,3) = \answer{-1}$ and $A(3,1) = \answer{0}$.
\end{question}
