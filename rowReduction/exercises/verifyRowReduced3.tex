\begin{question} The matrix
  \[M = \begin{bmatrix}
1 & 2 & 3 & 4\\
0 & 1 & 5 & 7\\
0 & 0 & 0 & 1
\end{bmatrix}\]
is
\begin{multipleChoice}
  \choice{not in row echelon form.}
  \choice[correct]{in row echelon form.}
\end{multipleChoice}

\begin{question}
  The matrix $M$ is
  \begin{multipleChoice}
    \choice[correct]{not in reduced row echelon form.}
    \choice{in reduced row echelon form.}
  \end{multipleChoice}
\end{question}
\end{question}
