\begin{problem}
  
  Let 
  \[A=\begin{bmatrix}
      1 & 1 & 0 \\
      1 &1 &0 
    \end{bmatrix}\]
  be a matrix. Find a basis $\mathcal{B}$ of the null space of the matrix $A$.
  \[
    \mathcal{B} = \left\{
        \begin{bmatrix}
          1  \\
          \answer{-1}  
        \end{bmatrix}
      \right\}
  \]
 \end{problem}
